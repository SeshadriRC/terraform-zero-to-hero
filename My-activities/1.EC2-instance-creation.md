## Creating key pair from Scratch

```yaml
terraform {
  required_providers {
    aws = {
      source  = "hashicorp/aws"
      version = "6.38.0"
    }
  }
}

provider "aws" {
  region = "ap-south-1"
}

# Generate RSA private key automatically
resource "tls_private_key" "xfusion_key" {
  algorithm = "RSA"
  rsa_bits  = 2048
}

# Create AWS key pair using generated public key
resource "aws_key_pair" "xfusion_kp" {
  key_name   = "first-ec2"
  public_key = tls_private_key.xfusion_key.public_key_openssh
}
# Fetch default VPC
data "aws_vpc" "default" {
  default = true
}

# Fetch default security group
data "aws_security_group" "default" {
  name   = "default"
  vpc_id = data.aws_vpc.default.id
}

# Create EC2 instance
resource "aws_instance" "first_ec2" {
  ami                    = "ami-0317b0f0a0144b137"
  instance_type          = "t2.micro"
  key_name               = aws_key_pair.xfusion_kp.key_name
  vpc_security_group_ids = [data.aws_security_group.default.id]

  tags = {
    Name = "xfusion-ec2"
  }
}
```

## Assigning the existing configurations

  required_providers {
    aws = {
      source  = "hashicorp/aws"

    }
  }

}

# Fetch default VPC
data "aws_vpc" "default" {
  default = true
}

# Fetch default security group
data "aws_security_group" "default" {
  name   = "default"
  vpc_id = data.aws_vpc.default.id
}

# Create EC2 instance
resource "aws_instance" "terraform-ec2" {

  vpc_security_group_ids = [data.aws_security_group.default.id]

  }
